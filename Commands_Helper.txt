# A tag is a named pointer to a specific commit ‚Äî usually used for releases:
# There are two types:
#  1. Lightweight tag   			- Just a name pointing to a commit (like a branch that never moves).
#  2. Annotated tag (recommended)   - Stored as a full object with: tag name, message, author, date, optional GPG signature

# List / show all tags

>  git tag
>  git tag -l "v1.*"         # List tags matching a pattern
>  git show v1.0             # Show tag details


# Create lightweight tag

>  git tag v1.0


# Create annotated tag (recommended)

>  git tag -a v1.0 -m "Release version 1.0"


#  Tag a specific commit

>  git tag -a v1.0 a1b2c3d


#  Push tags to remote

>  git push origin v1.0
>  git push --tags


#  Checkout tag commit (detached head)

>  git checkout v1.0
>  git checkout -b hotfix-1.0 v1.0   # Create a branch from a tag (safe way!)


#  Deleting and updating tags

>  git tag -d v1.0                       # Delete local tag
>  git push origin --delete v1.0         # Delete tag from remote


#  Replace an existing tag

>  git tag -d v1.0                       # Delete locally:
>  git tag -a v1.0 -m "corrected tag"    # Recreate
>  git push origin v1.0 --force          # Force-push tag:


#  Create signed GPG tags (secure releases)

>  git tag -s v1.0 -m "Signed release"
>  git tag -v v1.0                       # Verify signature:

# Diff between two versions

>  git diff v1.0..v2.0                   # Diff between two versions
>  git log  v1.0..v1.1 --oneline         # Show commits between tags


#  Archive code for a tag

>  git archive -o release-v1.0.zip v1.0


***** Practical tag workflow example ****

>  git tag -a v2.3.0 -m "Release 2.3.0"   # 1. Create release tag
>  git push origin v2.3.0                 # 2. Push it
>  git checkout -b hotfix-2.3 v2.3.0      # 3. Create hotfix from this version
>  git tag -a v2.3.1 -m "Hotfix for 2.3"  # 4. After finishing hotfix, tag again
